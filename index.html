The Assignment¶

Take a ZIP file) of images and process them, using a library built into python that you need to learn how to use. A ZIP file takes several different files and compresses them, thus saving space, into one single file. The files in the ZIP file we provide are newspaper images (like you saw in week 3). Your task is to write python code which allows one to search through the images looking for the occurrences of keywords and faces. E.g. if you search for "pizza" it will return a contact sheet of all of the faces which were located on the newspaper page which mentions "pizza". This will test your ability to learn a new (library), your ability to use OpenCV to detect faces, your ability to use tesseract to do optical character recognition, and your ability to use PIL to composite images together into contact sheets.

Each page of the newspapers is saved as a single PNG image in a file called images.zip. These newspapers are in english, and contain a variety of stories, advertisements and images. Note: This file is fairly large (~200 MB) and may take some time to work with, I would encourage you to use small_img.zip for testing.

# main method for zipfile unpacking and secondary method calling

import PIL
from PIL import Image
from zipfile import ZipFile

def main():
    f_name = 'readonly/images.zip'
    zip = ZipFile(f_name, 'r')
    f_list = zip.infolist()
    n_list = zip.namelist()

    keyword = input('Keyword to search for: ')
    
    for i in range(len(f_list)):
        image = Image.open(zip.open(f_list[i]))
        fh_name = 'readonly/temp.png'
        image.save(fh_name)
        print('\nChecking in file '+ n_list[i]+ " . . .")
        if key_present(fh_name, keyword):
            print('Keyword found in file, extracting faces . . .')
            face_detect(fh_name)
        else:
            print('Keyword not found!')
    print('\nDONE!')

# method for converting image data to text and checking if keyword is present

import cv2
import pytesseract
import PIL
from PIL import Image
import string

def key_present(temp, k):
    img = Image.open(temp)
    gray = cv2.cvtColor(cv2.imread(temp), cv2.COLOR_BGR2GRAY)
    
    ret, thresh1 = cv2.threshold(gray, 0, 255, cv2.THRESH_OTSU | cv2.THRESH_BINARY_INV)
    rect_kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (24, 24))
    dilation = cv2.dilate(thresh1, rect_kernel, iterations = 1) 
    contours, hierarchy = cv2.findContours(dilation, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_NONE)

    im2 = gray.copy()
    print('Converting file data to text and checking . . .')

    for cnt in contours: 
        x, y, w, h = cv2.boundingRect(cnt)
    
        crop = im2[y:y+h, x:x+w]
        text = pytesseract.image_to_string(crop)
        t_text = ""
        
        for ch in text.lower():
            if ch in string.ascii_lowercase:
                t_text += ch
        if k.lower() in t_text:
            return True
        
    return False

# methods for face detection and display using contact sheet

import cv2
import PIL
from PIL import Image

# creating contact sheet
def contact_sheet(ct):
    c_sheet = PIL.Image.new('RGB', (125*ct, 125))
    return c_sheet

# detecting faces and pasting into contact sheet
def face_detect(temp_f):
    face_cascade = cv2.CascadeClassifier('readonly/haarcascade_frontalface_default.xml')

    img = cv2.imread(temp_f)
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    faces = (face_cascade.detectMultiScale(gray, 1.235, 6))
    if len(faces) != 0:
        ct = contact_sheet(len(faces))
    
        image = Image.open(temp_f).convert('RGB')
        x = 0
    
        for face in faces:
            box = [face[0],face[1],face[0]+face[2],face[1]+face[3]]
            img = image.crop(box).resize((125, 125), PIL.Image.LANCZOS)
            ct.paste(img,(x,0))
            x += 125
            
        display(ct)
    else:
        print('There were no faces detected in this file!')
